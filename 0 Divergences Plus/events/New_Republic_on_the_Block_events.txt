country_event = { 
	id = 90820751 # Great powers get assigned a flag and the GP monarchies variable is changed
	title = "EVTNAME90820751"
	desc = "EVTDESC990820751"
	picture = "suppressing_republicanism"
	
	trigger = {
		is_greater_power = yes
		or = {
			government = absolute_monarchy
			government = prussian_constitutionalism
			government = hms_government
		}
		not = { has_country_flag = great_power_monarchy }
	}
	
	mean_time_to_happen = { days = 1 }
	
	option = {
		name = "EVTOPTA90820751"
		set_country_flag = great_power_monarchy
		change_variable = {
			which = great_power_monarchy
			value = 1.1
		}
	}
}
